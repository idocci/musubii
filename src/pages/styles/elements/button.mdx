# Button

## Basic

export const PreviewButtonBasic = () => {
  const contents = items
    .map(
      item =>
        `<button class="button is-plain ${item.role}">${item.text}</button>`
    )
    .join("")
    .replace(/\s+/g, " ")
    .replace(/\s\"/g, '"')
  const formattedCode = beautify.html(contents, beautifyHtmlOptions)
  return (
    <div className="demo-box is-preview">
      <div className="demo-box is-line">
        <div
          className="box is-flex is-space-row-xs is-space-column-xs"
          dangerouslySetInnerHTML={{ __html: formattedCode }}
        ></div>
      </div>
      <DemoPre language="html" code={formattedCode} />
    </div>
  )
}

<PreviewButtonBasic />
